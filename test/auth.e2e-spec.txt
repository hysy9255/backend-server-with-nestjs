      const res = await request(app.getHttpServer())

        .send({
          email: 'owner@example.com',
          password,
          role: 'Owner',
        })

        email: 'owner@example.com',
        password,
        role: 'Owner',
      });

        .post('/api/auth/login')
        .send({
          email: 'owner@example.com',
          password,
        })
        .expect(200);
